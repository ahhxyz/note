使用__构造函数模式__和__原型模式__混合的方式来实现面向对象

	//使用构造函数来定义类的属性及属性的初始化
	function Class(){
		this.property01 = '';
        this.property02 = '';
	}
    
    //使用原型来定义实例的共享成员，即方法和共享的成员变量
    Class.prototype.say = function(){
    	alert(this.property01);
    }
